let str = "rohit bhalchandra metakar";

function characterCount() {
    let arr = str.split("");
    let uniq = [...new Set(arr)];
    let result = {};

    for (let ele of uniq) {
        let count = str.split(ele).length - 1;
        result[ele] = count;
    }
    return result;
}
